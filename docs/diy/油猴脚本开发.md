```js
// @grant        GM_registerMenuCommand

```

## 菜单

```js
/*
  menu = {
    id: 0,
    name: '菜单名称',
    callback: 回调函数,
    key: '快捷键'
  }
*/
class Menu {
  constructor(menus) {
    this.menus = menus || []
    this.init()
  }

  add(menu) {
    this.menus.push(menu)
    this.init()
  }

  init() {
    this.menus.forEach(menu => {
      const id = GM_registerMenuCommand(menu.name, function () {
        menu.callback(id)
      }, menu.key)
    })
  }
}
```
